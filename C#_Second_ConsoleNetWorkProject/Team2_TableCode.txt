CREATE TABLE dbo.SignUp(
	cnt INT IDENTITY(1,1) PRIMARY KEY,
	id VARCHAR(50) NOT NULL,
	pw VARCHAR(50),
	name VARCHAR(50),
	ssn VARCHAR(50),
	phone VARCHAR(50)
);

CREATE TABLE dbo.TTable(
	cnt INT IDENTITY(1,1) PRIMARY KEY,
	attend VARCHAR(50) NOT NULL,
	name VARCHAR(50)
);

CREATE TABLE dbo.ETable(
	cnt INT IDENTITY(1,1) PRIMARY KEY,
	lie VARCHAR(50) NOT NULL
);
